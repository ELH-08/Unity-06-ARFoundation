import random

print("로또번호 1000개 출력:")
luck_No_Arr=[]
luck_No=[]

for i in range(5):

    while len(luck_No) < 7:
        num = int(random.randint(1, 45))
        if num in luck_No :
            continue
        else  :
            luck_No.append(num)

    luck_No_Arr.append(luck_No)
    print(luck_No_Arr[i])
    luck_No = []
